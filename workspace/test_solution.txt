/*
These tests assume the existence of the following symbol in the main package:

	// NewServer constructs and returns an http.Handler with all routes registered.
	func NewServer() http.Handler

The implementation under test MUST provide this function. All responses must
include the header:  Access-Control-Allow-Origin: *
*/

func ensureStaticDir(t *testing.T) (cleanup func()) {
	t.Helper()
	staticDir := "./static"
	if err := os.MkdirAll(staticDir, 0o755); err != nil {
		t.Fatalf("unable to create static directory: %v", err)
	}

	// Create a simple static file.
	filePath := filepath.Join(staticDir, "test.txt")
	content := []byte("hello world")
	if err := os.WriteFile(filePath, content, 0o644); err != nil {
		t.Fatalf("unable to write test static file: %v", err)
	}

	return func() {
		os.RemoveAll(staticDir)
	}
}

// helper to start the tested server and return httptest server & cleanup.
func newTestServer(t *testing.T) (*httptest.Server, func()) {
	handler := NewServer()
	ts := httptest.NewServer(handler)
	return ts, ts.Close
}

func TestStaticFileServingSuccess(t *testing.T) {
	cleanupFiles := ensureStaticDir(t)
	defer cleanupFiles()

	ts, closeServer := newTestServer(t)
	defer closeServer()

	resp, err := http.Get(ts.URL + "/static/test.txt")
	if err != nil {
		t.Fatalf("GET /static/test.txt failed: %v", err)
	}
	defer resp.Body.Close()

	if resp.StatusCode != http.StatusOK {
		t.Fatalf("expected status 200, got %d", resp.StatusCode)
	}

	body, _ := io.ReadAll(resp.Body)
	if string(body) != "hello world" {
		t.Errorf("unexpected body: %q", string(body))
	}

	contentType := resp.Header.Get("Content-Type")
	if !strings.Contains(contentType, "text/plain") {
		t.Errorf("expected Content-Type to contain 'text/plain', got %q", contentType)
	}

	if v := resp.Header.Get("Access-Control-Allow-Origin"); v != "*" {
		t.Errorf("expected CORS header '*', got %q", v)
	}
}

func TestStaticFileNotFound(t *testing.T) {
	ensureStaticDir(t) // we only need directory to exist

	ts, closeServer := newTestServer(t)
	defer closeServer()

	resp, err := http.Get(ts.URL + "/static/nonexistent.txt")
	if err != nil {
		t.Fatalf("GET /static/nonexistent.txt failed: %v", err)
	}
	defer resp.Body.Close()

	if resp.StatusCode != http.StatusNotFound {
		t.Fatalf("expected status 404, got %d", resp.StatusCode)
	}

	if v := resp.Header.Get("Access-Control-Allow-Origin"); v != "*" {
		t.Errorf("expected CORS header '*', got %q", v)
	}

	var payload map[string]interface{}
	if err := json.NewDecoder(resp.Body).Decode(&payload); err != nil {
		t.Fatalf("expected JSON body, got error: %v", err)
	}
	if _, ok := payload["error"]; !ok {
		t.Errorf("expected JSON payload with 'error' field, got %v", payload)
	}
}

func TestGreetingGETEndpoints(t *testing.T) {
	ts, closeServer := newTestServer(t)
	defer closeServer()

	for i := 1; i <= 5; i++ {
		path := "/api/greeting" + string('0'+i)
		resp, err := http.Get(ts.URL + path)
		if err != nil {
			t.Fatalf("GET %s failed: %v", path, err)
		}
		func() {
			defer resp.Body.Close()

			if resp.StatusCode != http.StatusOK {
				t.Errorf("%s: expected 200, got %d", path, resp.StatusCode)
			}

			if ct := resp.Header.Get("Content-Type"); !strings.Contains(ct, "application/json") {
				t.Errorf("%s: expected application/json, got %q", path, ct)
			}

			if v := resp.Header.Get("Access-Control-Allow-Origin"); v != "*" {
				t.Errorf("%s: expected CORS '*', got %q", path, v)
			}

			var body map[string]string
			if err := json.NewDecoder(resp.Body).Decode(&body); err != nil {
				t.Errorf("%s: cannot decode JSON: %v", path, err)
				return
			}
			want := "Hello from greeting" + string('0'+i)
			if body["message"] != want {
				t.Errorf("%s: expected message %q, got %q", path, want, body["message"])
			}
		}()
	}
}

	name := "Alice"
	for i := 1; i <= 5; i++ {

		payload := map[string]string{"name": name}

			t.Fatalf("POST %s failed: %v", path, err)

			if ct := resp.Header.Get("Content-Type"); !strings.Contains(ct, "application/json") {
				t.Errorf("%s: expected application/json, got %q", path, ct)
			}

			if v := resp.Header.Get("Access-Control-Allow-Origin"); v != "*" {
				t.Errorf("%s: expected CORS '*', got %q", path, v)
			}

			var body map[string]string
			if err := json.NewDecoder(resp.Body).Decode(&body); err != nil {
				t.Errorf("%s: cannot decode JSON: %v", path, err)

			}
			want := "Hello, " + name + " from greeting" + string('0'+i)
			if body["message"] != want {
				t.Errorf("%s: expected message %q, got %q", path, want, body["message"])
			}
		}()
	}
}

func TestConcurrentPerformance(t *testing.T) {
	cleanupFiles := ensureStaticDir(t)
	defer cleanupFiles()

	ts, closeServer := newTestServer(t)
	defer closeServer()

	const totalRequests = 100
	durations := make([]time.Duration, totalRequests)
	wg := sync.WaitGroup{}
	wg.Add(totalRequests)

	startAll := make(chan struct{})

	for i := 0; i < totalRequests; i++ {
		go func(idx int) {
			defer wg.Done()
			<-startAll // wait for the signal to start together

			var url string
			if idx%2 == 0 {
				url = ts.URL + "/static/test.txt"
			} else {
				endpoint := "/api/greeting" + string('0'+(idx%5)+1)
				url = ts.URL + endpoint
			}

			start := time.Now()
			resp, err := http.Get(url)
			if err != nil {
				t.Errorf("request error: %v", err)
				return
			}
			io.Copy(io.Discard, resp.Body)
			resp.Body.Close()
			durations[idx] = time.Since(start)
		}(i)
	}

	close(startAll) // unleash all goroutines
	wg.Wait()

	// Compute 95th percentile latency
	sort.Slice(durations, func(i, j int) bool { return durations[i] < durations[j] })
	index95 := int(math.Ceil(float64(totalRequests)*0.95)) - 1
	lat95 := durations[index95]

	if lat95 > 100*time.Millisecond {
		t.Errorf("95th percentile latency too high: %v (must be <= 100ms)", lat95)
	}
}
